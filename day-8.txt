package day8assesment;

public class Assignment6 {
	public static void printPascal(int k) {
		for (int line=1; line<=k;line++) {
			for(int b=0; b<=k -line;b++) {
				//print whitespaces for left spacing
				System.out.println(" ");
			}
			//variable used to represent C(line,i)
			int C=1;
			for(int a=1;a<=line;a++) {
				//print first value in a line is always 1
				System.out.println(C + " ");
				C =C*(line-a)/a;
			}
			//BY now,we are done with one row so a now line is required
			System.out.println();
		}
	}
	public static void main(String[] args) {
		int n=6;
		printPascal(n);
	}

}
